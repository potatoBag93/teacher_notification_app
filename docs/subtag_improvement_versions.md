## 📋 기본 설정: 축약형 메인카테고리 13개

```typescript
export type Category = 
  | '학습관리'    // 기존: 학습 + 숙제 + 준비물
  | '생활지도'    // 유지
  | '안전보건'    // 기존: 안전 + 건강
  | '인성교육'    // 기존: 인성 + 칭찬 + 주의
  | '창의예술'    // 기존: 창의 + 예술
  | '체육건강'    // 기존: 체육 (건강과 일부 통합)
  | '환경정보'    // 기존: 환경 + 정보
  | '행사활동'    // 기존: 행사
  | '상담지원'    // 기존: 상담
  | '학교알림'    // 기존: 알림
  | '특별교육'    // 계절/시기별 특별 교육
  | '가정연계'    // 가정과의 소통
  | '기타사항'    // 기존: 기타 + 일반
```
